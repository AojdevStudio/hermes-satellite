def create_mcp_server() -> "FastMCP":
    """Create the async task bridge MCP server."""
    try:
        from mcp.server.fastmcp import FastMCP
    except ImportError:
        logger.error("MCP SDK not available. Install with: pip install mcp")
        sys.exit(1)

    mcp = FastMCP(
        "hermes-async",
        instructions=(
            "Hermes Agent async task bridge. Submit tasks to Hermes Agent "
            "(running on a remote Mac Mini) and get results back asynchronously. "
            "Hermes has full tool access: terminal, file, web, search, skills, "
            "memory, delegation, and more. "
            "\n\n"
            "Workflow:\n"
            "1. Call hermes_submit with a prompt to start a task.\n"
            "2. Poll hermes_status until status is 'completed' or 'failed'.\n"
            "3. Call hermes_result to get the full response.\n"
            "4. Call hermes_respond with the original task_id to send a follow-up "
            "that resumes the same Hermes session.\n"
            "\n"
            "Tasks can take seconds to minutes depending on complexity. "
            "Hermes runs with full homelab access — terminal, SSH, Docker, "
            "file editing, web search, and all installed skills."
        ),
    )

    task_mgr = TaskManager()

    @mcp.tool()
    def hermes_result(task_id: str) -> str:
        """Get the full result of a completed (or failed) task.

        Only returns the result if the task is completed or failed.
        For running tasks, call hermes_status first.

        Args:
            task_id: The task_id from hermes_submit.

        Returns:
            JSON with the full Hermes response, session_id (for follow-ups),
            and any error message.
        """
        if not task_id:
            return json.dumps({"error": "task_id is required"})

        result = task_mgr.get_result(task_id)
        if not result:
            return json.dumps({"error": f"Task not found: {task_id}"})

        if result["status"] not in ("completed", "failed", "cancelled"):
            return json.dumps({
                "task_id": task_id,
                "status": result["status"],
                "message": "Task is still running. Call hermes_status to poll.",
            })

        return json.dumps(result)
